export async function getServerSideProps(context) {
  let id = context.params.id;
  let originalEIP;
  try {
    originalEIP = await readFile(
      path.join(process.cwd(), 'public', 'original-eips', `${id}.md`),
      'utf8'
    );
  } catch (e) {
    console.log(e);
    if (e.errno === -2) {
      return {
        redirect: {
          destination: '/404',
          permanent: false,
        },
      };
    }
  }
  let meta, con;
  if (originalEIP) {
    [, meta, ...con] = originalEIP.split('---');
  }
  meta = formatMeta(meta);
  try {
    let comEIP = await readFile(
      path.join(process.cwd(), 'content', 'en', `${id}.md`),
      'utf8'
    );
    Object.assign(meta, formatComEIP(comEIP));
  } catch (e) {
    console.log(e);
  }
  return {
    props: {
      meta,
      jsxData: con.toString(),
      id: id,
    },
  };
}
