export const flatten = (text, child) => {
  return typeof child === "string"
    ? text + child
    : React.Children.toArray(child.props.children).reduce(flatten, text);
};
export const HeadingRenderer = (props) => {
    let children = React.Children.toArray(props.children);
    let text = children.reduce(flatten, "");
    let slug = text.toLowerCase().replace(/[!?\s]/g, "-");
    
    return React.createElement(
        "h" + props.level,
        { id: slug, className: "eip-toc" },
        props.children
    );
};

"#ECECEC" : 'transparent'} zIndex={2}>
              <ul id="sideMenu" className={detials.eipEnmu}>

      <EmailSubscribe />
    </Box>

  </Container>
</>);
}

export async function getServerSideProps(context) {
  let id = context.params.id;
  let originalEIP;
  if (id.includes('.md')) {
    id = id.replace('.md', '');
  }
  if (id.includes('#')) {
    id = id.split('#')[0];
  }
  try {
    originalEIP = await readFile(
      path.join(process.cwd(), 'public', 'original-eips', `${id}.md`),
      'utf8'
    );
  } catch (e) {
    console.log(e);
    if (e.errno === -2) {
      return {
        redirect: {
          destination: '/404',
          permanent: false,
        },
      };
    }
  }
  let meta, con, sideMenu;
  if (originalEIP) {
    [, meta, ...con] = originalEIP.split('---');
  }
  meta = formatMeta(meta);
  sideMenu = getHeader(con.toString());
  try {
    let comEIP = await readFile(
      path.join(process.cwd(), 'content', 'en', `${id}.md`),
      'utf8'
    );
    Object.assign(meta, formatComEIP(comEIP));
  } catch (e) {
    console.log(e);
  }
  return {
    props: {
      meta,
      jsxData: con.toString(),
      id: id,
      sideMenu,
    },
  };
}
